import { AxiosModel } from 'AxiosModel';

// Test doc user
const DEMO_USER = {
  id: 4,
  last_name: 'Holt',
  first_name: 'Eve',
  email: 'eve.holt@reqres.in',
  avatar: 'https://s3.amazonaws.com/uifaces/faces/twitter/marcoramires/128.jpg'
};

// Setting for PersonModel
const PARAMS = {
  generateProperty: true,
  base: 'https://reqres.in/api',
  path: '/users',
  properties: {
    avatar: '',
    email: '',
    last_name: '',
    first_name: ''
  }
};

class PersonModel extends AxiosModel {
  constructor(data = {}) {
    super();
    this._data = data;
  }

  static get _opts() {
    return PARAMS;
  }

  // Add new getter/setter merge name (optional)
  get name() {
    return `${this._data.first_name} ${this._data.last_name}`
  }

  set name({first_name, last_name}) {
    this._data = first_name + ' new'
    this._data = last_name + ' new'
  }

  // Custome getter email (optional)
  get email() {
    return this._data.email ? this._data.email : '–'
  }
};
